Vargas::ofasta Class Reference

#include <fasta.h>

Detailed Description

FASTA file writer.

Usage:

#include "fasta.h"
{
Vargas::ofasta out("output.fa");
out.char_per_line(5);
out.write("seq1", "ACGTCACCT");
out.write("seq2", "ACGT");
// output.fa:
// >seq1
// ACGTC
// ACCT
// >seq2
// ACGT
}
{
Vargas ofasta out();
out.char_per_line(3);
out.write("seq1", "ACGTCTCTC");
// stdout:
// >seq1\nACG\nTCT\nCTC\n
}
