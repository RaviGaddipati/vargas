387  inline void _package_reads() {
388  _packaged_reads.resize(read_len);
389  if (_reads.size() > num_reads) throw std::range_error("Too many reads for batch size.");
390 
391  // allocate memory
392  uchar **pckg = (uchar **) malloc(read_len * sizeof(uchar *));
393  for (int i = 0; i < read_len; ++i) {
394  pckg[i] = (uchar *) malloc(num_reads * sizeof(uchar));
395  }
396 
397  // Interleave reads
398  // For each read (read[i] is in _packaged_reads[0..n][i]
399  for (size_t r = 0; r < _reads.size(); ++r) {
400 
401  if (_reads.at(r).read_num.size() > read_len) throw std::range_error("Read too long for batch size.");
402 
403  // Put each base in the appropriate vector element
404  for (size_t p = 0; p < _reads[r].read_num.size(); ++p) {
405  pckg[p][r] = _reads[r].read_num[p];
406  }
407 
408  // Pad the shorter reads
409  for (size_t p = _reads[r].read_num.size(); p < read_len; ++p) {
410  pckg[p][r] = Base::N;
411  }
412  }
413 
414  // Pad underful batches
415  for (size_t r = _reads.size(); r < num_reads; ++r) {
416  for (size_t p = 0; p < read_len; ++p) {
417  pckg[p][r] = Base::N;
418  }
419  }
420 
421  // Load into vectors
422  for (int i = 0; i < read_len; ++i) {
423  _packaged_reads[i] = simdpp::load(pckg[i]);
424  }
425 
426  // Free memory
427  for (int i = 0; i < read_len; ++i) {
428  free(pckg[i]);
429  }
430  free(pckg);

432  }
