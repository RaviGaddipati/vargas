276  inline void _package_reads() {
277  _packaged_reads.resize(read_len);
278  if (_reads.size() > num_reads) throw std::range_error("Too many reads for batch size.");
279 
280  // allocate memory
281  uchar **pckg = (uchar **) malloc(read_len * sizeof(uchar *));
282  for (int i = 0; i < read_len; ++i) {
283  pckg[i] = (uchar *) malloc(num_reads * sizeof(uchar));
284  }
285 
286  // Interleave reads
287  // For each read (read[i] is in _packaged_reads[0..n][i]
288  for (size_t r = 0; r < _reads.size(); ++r) {
289 
290  if (_reads.at(r).read_num.size() > read_len) throw std::range_error("Read too long for batch size.");
291 
292  // Put each base in the appropriate vector element
293  for (size_t p = 0; p < _reads[r].read_num.size(); ++p) {
294  pckg[p][r] = _reads[r].read_num[p];
295  }
296 
297  // Pad the shorter reads
298  for (size_t p = _reads[r].read_num.size(); p < read_len; ++p) {
299  pckg[p][r] = Base::N;
300  }
301  }
302 
303  // Pad underful batches
304  for (size_t r = _reads.size(); r < num_reads; ++r) {
305  for (size_t p = 0; p < read_len; ++p) {
306  pckg[p][r] = Base::N;
307  }
308  }
309 
310  // Load into vectors
311  for (int i = 0; i < read_len; ++i) {
312  _packaged_reads[i] = simdpp::load(pckg[i]);
313  }
314 
315  // Free memory
316  for (int i = 0; i < read_len; ++i) {
317  free(pckg[i]);
318  }
319  free(pckg);
320 
321  }

323  };
