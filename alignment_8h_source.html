221  void align_into(ReadBatch<num_reads, CellType> reads,
222  Graph::FilteringIter begin,
223  Graph::FilteringIter end,
224  std::vector<Alignment> &aligns) {
225  using namespace simdpp;
226 
227  max_score = ZERO_CT;
228  sub_score = ZERO_CT;
229  max_pos = std::vector<uint32_t>(num_reads);
230  _seed seed(read_len);
231  std::unordered_map<uint32_t, _seed> seed_map;
232  for (auto gi = begin; gi != end; ++gi) {
233  _get_seed(gi.incoming(), seed_map, &seed);
234  if ((*gi).is_pinched()) seed_map.clear();

245  sub, sub_pos[i], sub_count[i], corflag[i]);

383  _seed _fill_node(const Graph::Node &n,
384  const ReadBatch<num_reads, CellType> &reads,
385  const _seed *s) {
386 
387  _seed nxt(read_len); // Seed for next node
388  auto node_seq = n.seq().data(); // Raw pointer faster than at()
389  const CellType<num_reads> *read_ptr = reads.data();
390  size_t seq_size = n.seq().size();
391  uint32_t node_origin = n.end() - seq_size;
392 
393  // top left corner
394  _fill_cell_rzcz(read_ptr[0], node_seq[0], s);
395  _fill_cell_finish(0, 0, node_origin, reads.reads());
396 
397  // top row
398  for (uint32_t c = 1; c < seq_size; ++c) {
399  _fill_cell_rz(read_ptr[0], node_seq[c], c);
400  _fill_cell_finish(0, c, node_origin, reads.reads());
401  }
402 
403  nxt.S_col[0] = S_curr[seq_size - 1];
404 
405  // Rest of the rows
406  for (uint32_t r = 1; r < read_len; ++r) {
407  // Swap the rows we are filling in. The previous row/col becomes what we fill in.
408  swp_tmp = S_prev;
409  S_prev = S_curr;
410  S_curr = swp_tmp;
411 
412  swp_tmp = D_prev;
413  D_prev = D_curr;
414  D_curr = swp_tmp;
415 
416  swp_tmp = I_prev;
417  I_prev = I_curr;
418  I_curr = swp_tmp;
419 
420  // first col
421  _fill_cell_cz(read_ptr[r], node_seq[0], r, s);
422  _fill_cell_finish(r, 0, node_origin, reads.reads());
423 
424  // Inner grid
425  for (uint32_t c = 1; c < seq_size; ++c) {
426  _fill_cell(read_ptr[r], node_seq[c], r, c);
427  _fill_cell_finish(r, c, node_origin, reads.reads());
428  }
429 
430  nxt.S_col[r] = S_curr[seq_size - 1];
431 
432  }
433 
434  // origin vector of what is now I_curr
435  nxt.I_col = (Ia->data() == I_curr) ? *Ia : *Ib;
436  return nxt;
437  }

516  __INLINE__
517  void _D(const uint32_t &col,
518  const CellType<num_reads> &Dp,
519  const CellType<num_reads> &Sp) {
520  using namespace simdpp;
521 
522  // D(i,j) = D(i-1,j) - gap_extend
523  // Dp is D_prev[col], 0 for row=0
524  D_curr[col] = sub_sat(Dp, _gap_extend);
525  // tmp = S(i-1,j) - ( gap_open + gap_extend)
526  // Sp is S_prev[col], 0 for row=0
527  tmp = sub_sat(Sp, _gap_extend + _gap_open);
528  // D(i,j) = max{ D(i-1,j) - gap_extend, S(i-1,j) - ( gap_open + gap_extend) }
529  D_curr[col] = max(D_curr[col], tmp);
530  }

559  __INLINE__
560  void _M(uint32_t col,
561  const CellType<num_reads> &read,
562  const Base &ref,
563  const CellType<num_reads> &Sp) {
564  using namespace simdpp;
565 
566  Ceq = ZERO_CT;
567  Cneq = ZERO_CT;
568 
569  if (ref != Base::N) {
570  // Set all mismatching pairs to _mismatch
571  tmp = cmp_neq(read, ref);
572  Cneq = tmp & _mismatch;
573  // If the read base is Base::N, set to 0 (Ceq)
574  tmp = cmp_eq(read, Base::N);
575  Cneq = blend(Ceq, Cneq, tmp);
576 
577  // b is not N, so all equal bases are valid
578  tmp = cmp_eq(read, ref);
579  Ceq = tmp & _match;
580  }
581 
582  // Sp is S_prev[col - 1], 0 for row=0
583  // Seed->S_col[row - 1] for col=0
584  S_curr[col] = add_sat(Sp, Ceq); // Add match scores
585  S_curr[col] = sub_sat(S_curr[col], Cneq); // Subtract mismatch scores
586  }

598  __INLINE__
599  void _fill_cell_finish(const uint32_t &row,
600  const uint32_t &col,
601  const uint32_t &node_origin,
602  const std::vector<Read> &reads) {
603  using namespace simdpp;
604 
605  uint32_t curr = node_origin + col + 1; // absolute position
606 
607  //S(i,j) = max{ D(i,j), I(i,j), S(i-1,j-1) + C(s,t) }
608  S_curr[col] = max(D_curr[col], S_curr[col]);
609  S_curr[col] = max(I_curr[row], S_curr[col]);
610 
611  // Check for new high scores
612  tmp = S_curr[col] > max_score;
613  NativeT max_elem;
614  if (reduce_or(tmp)) {
615  max_score = max(max_score, S_curr[col]);
616  for (uchar i = 0; i < num_reads; ++i) {
617  // Check if the i'th elements MSB is set
618  if (extract(i, tmp)) {
619  max_elem = extract(i, max_score);
620  // If old max is larger than old sub_max, and if its far enough away
621  if (max_elem > extract(i, sub_score) && curr < max_pos[i] - read_len) {
622  insert(max_elem, i, sub_score);
623  sub_pos[i] = max_pos[i];
624  sub_count[i] = max_count[i];
625  }
626  max_pos[i] = curr;
627  max_count[i] = 0;
628  // Invalidate previous best match
629  if (corflag[i] == 1) corflag[i] = 0;
630  }
631  }
632  }
633 
634  // Check for equal max score. If we set a new high score this will set the count to 1
635  tmp = cmp_eq(S_curr[col], max_score);
636  if (reduce_or(tmp)) {
637  for (uchar i = 0; i < num_reads; ++i) {
638  // Check if the i'th elements MSB is set
639  if (extract(i, tmp)) {
640  ++max_count[i];
641  if (reads[i].end_pos == curr) corflag[i] = 1;
642  }
643  }
644  }
645 
646  // new second best score
647  tmp = S_curr[col] > sub_score;
648  if (reduce_or(tmp)) {
649  for (uchar i = 0; i < num_reads; ++i) {
650  // Check if the i'th elements MSB is set, and if far enough
651  if (extract(i, tmp) && max_pos[i] < curr - read_len) {
652  //TODO add -(max_score/sub_score) term
653  insert(extract(i, S_curr[col]), i, sub_score);
654  sub_pos[i] = curr;
655  sub_count[i] = 0;
656  // Invalidate previous second best match
657  if (corflag[i] == 2) corflag[i] = 0;
658  }
659  }
660  }
661 
662  // Repeat sub score
663  tmp = cmp_eq(S_curr[col], sub_score);
664  if (reduce_or(tmp)) {
665  for (uchar i = 0; i < num_reads; ++i) {
666  // Check if the i'th elements MSB is set
667  if (extract(i, tmp)) {
668  ++sub_count[i];
669  if (reads[i].end_pos == curr && corflag[i] == 0) corflag[i] = 2;
670  }
671  }
672  }
673  }
