234  __INLINE__ void _package_reads(const std::vector<std::vector<Base>> &_reads) {
235  assert(_reads.size() <= SIMDPP_FAST_INT8_SIZE);
236  // Interleave reads

238  for (size_t r = 0; r < _reads.size(); ++r) {

586  void _fill_node(const Graph::Node &n,
587  const AlignmentGroup &read_group,
588  const _seed *s,
589  _seed *nxt) {
590 
591  // Empty nodes repersents deletions
592  if (n.seq().size() == 0) {
593  nxt->I_col = s->I_col;
594  nxt->S_col = s->S_col;
595  return;
596  }
597 
598  assert(n.seq().size() <= _max_node_len);
599 
600  const Base *node_seq = n.seq().data();
601  const simdpp::uint8<SIMDPP_FAST_INT8_SIZE> *read_ptr = read_group.data();
602  const size_t seq_size = n.seq().size();
603  const size_t node_origin = n.end() - seq_size + 2;
604 
605  // top left corner
606  _fill_cell_rzcz(read_ptr[0], node_seq[0], s);
607  _fill_cell_finish(0, 0, node_origin);
608 
609  // top row
610  for (uint32_t c = 1; c < seq_size; ++c) {
611  _fill_cell_rz(read_ptr[0], node_seq[c], c);
612  _fill_cell_finish(0, c, node_origin);
613  }
614 
615  nxt->S_col[0] = _S_curr[seq_size - 1];
616 
617  // Rest of the rows
618  for (uint32_t r = 1; r < _read_len; ++r) {
619  // Swap the rows we are filling in. The previous row/col becomes what we fill in.
620  _swp_tmp0 = _S_prev;
621  _S_prev = _S_curr;
622  _S_curr = _swp_tmp0;
623 
624  _swp_tmp0 = _D_prev;
625  _D_prev = _D_curr;
626  _D_curr = _swp_tmp0;
627 
628  _swp_tmp0 = _I_prev;
629  _I_prev = _I_curr;
630  _I_curr = _swp_tmp0;
631 
632  // first col
633  _fill_cell_cz(read_ptr[r], node_seq[0], r, s);
634  _fill_cell_finish(r, 0, node_origin);
635 
636  // Inner grid
637  for (uint32_t c = 1; c < seq_size; ++c) {
638  _fill_cell(read_ptr[r], node_seq[c], r, c);
639  _fill_cell_finish(r, c, node_origin);
640  }
641 
642  nxt->S_col[r] = _S_curr[seq_size - 1];
643 
644  }
645 
646  // origin vector of what is now _I_curr
647  nxt->I_col = _Ia->data() == _I_curr ? *_Ia : *_Ib;
648  }

810  __INLINE__ __UNROLL__
811  void _fill_cell_finish(const uint32_t &row,
812  const uint32_t &col,
813  const uint32_t &node_origin) {
814  using namespace simdpp;
815 
816 
817 
818  // S(i,j) = max{ D(i,j), I(i,j), S(i-1,j-1) + C(s,t) }
819  _S_curr[col] = max(_D_curr[col], _S_curr[col]);
820  _S_curr[col] = max(_I_curr[row], _S_curr[col]);
821

823 
824  _tmp0 = _S_curr[col] > _max_score;
825  if (extract_bits_any(_tmp0)) {
826  // Check for new or equal high scores
827  _max_score = max(_S_curr[col], _max_score);
828  for (int i = 0; i < SIMDPP_FAST_INT8_SIZE; ++i) {
829  if (_tmp0_ptr[i]) {

831  if (_curr_pos > _max_pos[i] + _read_len) {
832  _sub_score_ptr[i] = _max_score_ptr[i];
833  _sub_pos[i] = _max_pos[i];
834  _sub_count[i] = _max_count[i];
835  _cor_flag[i] = (_cor_flag[i] == 1) * 2;
836  }
837  _max_count[i] = 1;
838  _cor_flag[i] = (_cor_flag[i] == 2) * 2;
839  }
840  }
841  }
842 
843  _tmp0 = cmp_eq(_S_curr[col], _max_score);
844  if (extract_bits_any(_tmp0)) {
845  // Check for equal max score.
